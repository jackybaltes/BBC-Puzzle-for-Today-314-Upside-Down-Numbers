Simple Implementation

This implementation is cumbersome. It uses a lot of text to achieve a simple goal

Furthermore, not every digit maps to itself when reversed. For example, 6 -> 9 and vice versa

So we need to return more than just a boolean

start = 1
end = 10000000

count_reversible_and_same = 0
count_reversible = 0

def is_reversible_and_same( inum):
  """
  returns 0 if number is not reversible, 1 if reversible and not same, or 2 if reversible and same
  """
  # Version 1
  # digits = [ ord(s) - ord('0') for s in str(num) ]
  
  num = inum
  # Version 2
  digits = []
  while( num > 0 ):
    d = num % 10
    #digits = [ d ] + digits
    digits.append(d)
    num = num // 10

  digit_map = { 0: 0, 1:1, 2:2, 5:5, 6:9, 8:8, 9:6 }

  reversible = 1

  val_reversed_num = 0

  for d in digits:
    if d in digit_map:
      val_reversed_num = val_reversed_num * 10 + digit_map[d]
    else:
      reversible = 0
      break

  if reversible > 0:
    if inum == val_reversed_num:
      reversible = 2

  #print('check', inum, digits, reversible, val_reversed_num)
  return reversible

for num in range(start, end+1):
  chk = is_reversible_and_same(num)
  if chk >= 2:
    count_reversible_and_same = count_reversible_and_same + 1
  if chk >= 1:
    count_reversible = count_reversible + 1

print( 'Reversible numbers', count_reversible, 'and same', count_reversible_and_same )
